function pingServer() {
  return new Promise((resolve, reject) => {
    const isOnline = Math.random() > 0.7;
    setTimeout(() => {
      isOnline ? resolve("ğŸŸ¢ Server is online!") : reject("ğŸ”´ Server is still offline...");
    }, 1000);
  });
}

async function checkServerWithRetry(maxTries = 5) {
  for (let i = 1; i <= maxTries; i++) {
    try {
      console.log(`ğŸ” Attempt ${i}...`);
      const status = await pingServer();
      console.log(status);
      return;
    } catch (err) {
      console.log(err);
    }
  }
  console.log("âŒ Failed to connect after multiple attempts.");
}
